```python
async def extract_skill_from_video(video_url: str) -> str:
    """从视频自动创建技能"""
    
    # 1. 分析视频
    learner = VideoLearner()
    result = await learner.analyze(video_url)
    
    # 2. 识别技能机会
    skill_opportunity = {
        "name": generate_skill_name(result),
        "category": classify_category(result),
        "description": result.summary,
        "commands": extract_commands(result),
        "examples": extract_examples(result),
        "best_practices": extract_best_practices(result)
    }
    
    # 3. 生成 SKILL.md
    skill_content = generate_skill_md(skill_opportunity)
    
    # 4. 保存技能文件
    skill_path = f"skills/{skill_opportunity['name']}/SKILL.md"
    save_file(skill_path, skill_content)
    
    return skill_path
```

### 技能质量评估

```python
def assess_skill_quality(video_result) -> dict:
    """评估视频是否适合提取技能"""
    
    scores = {
        # 内容密度
        "content_density": len(video_result.transcript) / video_result.duration,
        
        # 可操作性 (是否有具体命令/步骤)
        "actionability": len(video_result.commands) / 10,
        
        # 讲解清晰度
        "clarity": assess_explanation_clarity(video_result),
        
        # 技术深度
        "depth": assess_technical_depth(video_result),
        
        # 实用性
        "practicality": len(video_result.action_items) / 5
    }
    
    overall_score = sum(scores.values()) / len(scores)
    
    return {
        "scores": scores,
        "overall": overall_score,
        "recommendation": "创建技能" if overall_score > 0.7 else "仅学习"
    }
```

```python
async def learn_from_playlist(playlist_url: str, max_videos: int = 10):

    # 生成综合学习报告
    return {
        'total_videos': len(videos),
        'analyzed': len(results),
        'syllabus': generate_syllabus(results),
        'knowledge_graph': build_knowledge_graph(results),
        'skills': extract_skills_from_series(results)
    }
```

### 并发处理

```python
async def batch_analyze(urls: list[str], max_concurrent: int = 3):
    """并发分析多个视频"""
    
    semaphore = asyncio.Semaphore(max_concurrent)
    learner = VideoLearner()
    
    async def analyze_with_limit(url: str):
        async with semaphore:
            try:
                return await learner.analyze(url)
            except Exception as e:
                print(f"分析失败 {url}: {e}")
                return None
    
    results = await asyncio.gather(*[
        analyze_with_limit(url) for url in urls
    ])
    
    return [r for r in results if r is not None]
```

```python
async def extract_multilingual_transcript(video_id: str):
    """提取多语言字幕"""
    from youtube_transcript_api import YouTubeTranscriptApi
    
    # 获取可用字幕列表
    transcript_list = YouTubeTranscriptApi.list_transcripts(video_id)
    
    transcripts = {}
    for transcript in transcript_list:
        transcripts[transcript.language_code] = transcript.fetch()
    
    return transcripts
```

## 集成到学习循环

### 每日视频学习任务

```python
async def daily_video_learning():
    """每日视频学习工作流"""
    
    learner = VideoLearner()
    
    # 1. 发现待学习视频
    videos_to_learn = await discover_videos_from_sources()
    
    # 2. 筛选高质量内容
    filtered = filter_by_quality(videos_to_learn)
    
    # 3. 批量分析 (每天最多5个)
    for video in filtered[:5]:
        result = await learner.analyze(video['url'])
        
        # 4. 提取技能
        quality = assess_skill_quality(result)
        if quality['recommendation'] == "创建技能":
            skill_path = await extract_skill_from_video(video['url'])
            print(f"创建新技能: {skill_path}")
        
        # 5. 保存学习笔记
        save_learning_notes(result)
        
        # 6. 记录学习
        log_learning(video, result)
    
    # 7. 生成日报
    return generate_daily_learning_report()
```
