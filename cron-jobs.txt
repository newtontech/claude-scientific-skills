# Cron job configuration for Claude Scientific Skills daily contributions
# Save this to your crontab with: crontab -e

# Daily contribution check at 9:00 AM
0 9 * * * cd /home/yhm/.openclaw/workspace/claude-scientific-skills && /usr/bin/git fetch upstream && /usr/bin/git rebase upstream/main && /usr/bin/git push origin main 2>&1 | logger -t "claude-skills-sync"

# Weekly skill improvement check (every Sunday at 10:00 AM)
0 10 * * 0 cd /home/yhm/.openclaw/workspace/claude-scientific-skills && /usr/bin/gh pr list --repo K-Dense-AI/claude-scientific-skills --state open 2>&1 | logger -t "claude-skills-pr-check"

# Monthly repository update (1st of every month at 8:00 AM)
0 8 1 * * cd /home/yhm/.openclaw/workspace/claude-scientific-skills && /usr/bin/git remote prune origin 2>&1 | logger -t "claude-skills-cleanup"
